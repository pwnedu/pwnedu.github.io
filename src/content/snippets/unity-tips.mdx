## Scriptable Settings Singleton custom class example

```swift
    [FilePath("Config/MySettings.asset", FilePathAttribute.Location.PreferencesFolder)]
    public class MySettings : ScriptableSingleton<MySettings>
    {
        [SerializeField] private bool _myThingEnabled = true;

        public bool MyThingEnabled 
        { 
            get => _myThingEnabled;
            set 
            {
                _myThingEnabled = value;
                Save(true);
            } 
        }
    }
```
